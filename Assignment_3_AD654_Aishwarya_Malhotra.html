#Perform t-test for each pair of groups with Bonferroni correction
for i in range(len(groups)):
    for j in range(i+1, len(groups)):
        group1 = df[df['campaign_seen'] == groups[i]]['total_spending']
        group2 = df[df['campaign_seen'] == groups[j]]['total_spending']
        
        # t-Test
        t_statistic, p_value = ttest_ind(group1, group2)
        
        # Print results with Bonferroni correction
        print(f"T-test between {groups[i]} and {groups[j]}:")
        print(f"T-statistic: {t_statistic}")
        print(f"P-value: {p_value}")
        if p_value < alpha_bonferroni:
            print("Null hypothesis rejected (significant)")
        else:
            print("Null hypothesis not rejected (not significant)")
        print()

observed = [150, 120, 90, 80, 60]

# Expected frequencies
expected = [150, 125, 75, 75, 75]

chi_square = 0

for obs, exp in zip(observed, expected):
    chi_square += ((obs - exp) ** 2)/ exp

print("Chi - square value", chi_square)

Chi - square value 6.533333333333333

Interpretation of Results-

Chi-square statistics is measure of difference between observed and expected frequencies (O-E).

For each observed number in the table, subtract the corresponding expected number (O E).

Square the difference (O E).

Divide the squares obtained for each cell in the table by the expected number for that cell (O E) / E.

Sum all the values for (O E) / E. This is the chi-square statistic.
